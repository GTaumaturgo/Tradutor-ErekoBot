#include <Servo.h>
Servo Motor;
float cont;
float a;
float k;
float M;
int i;
float s;
float phase;
float angulo;
void setup(){
	Serial.begin(9600);
	Motor.attach(9);            // conecta o servo no pino 9
	//Motor.write(120);
	cont = 0;
	a = 0.5;
	k = 2;
	M = 8;
	i = 0;
	s = 7;
}
void loop(){
	delay(17);
	phase = i * (3.14159265 / 180);
	angulo = 2 * a * sin( 3.14159265 * k / M) * sin( phase + (2 * 3.14159265 * k / M) * (s + 1 / 2));
	Motor.write(((angulo) * 180/ 3.14159265) + 90 );
	i = i + 1;
	i = i % 360;
}
